\recipe{Owsianka z tofu}
\ingred{\begin{tabular}{lll}
\textbf{składniki na 4 osoby} \textbf{} & \textbf{cena w PLN}\\
tofu & 1 kostka & 4\\
cebula & 1 sztuka & 0,5\\
imbir & 4cm & 1\\
owsianka & ile trzeba & 1\\
oliwa z oliwek & dwie łyżki & 0,5\\
przyprawy\footnote{zioła prowansalskie, gałka muszkatołowa, kolendra, zielona pasta curry, ew. sól} &  do smaku & 0,5\\
szacunkowa & łączna kwota & 67 PLN\\
\end{tabular}}
\begin{enumerate}
\item ugotuj owsiankę
\item tofu odsącz i pokrój w kostkę
\item cebulę pokrój w półtalarki
\item do rozgrzanego woka wlej oliwę, wrzuć tofu (mieszaj bo lubi się przykleić), a gdy się zarumieni - cebulę i imbir
\item gdy cebula zacznie robić się miękka dodaj pół łyżeczki pasty curry i przyprawy
\item po kilku minutach zestaw z ognia i wymieszaj z owsianką 
\end{enumerate}
